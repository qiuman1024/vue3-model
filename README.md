# src目录结构

├── App.vue # 根组件
├── api # api接口
├── assets # 静态资源
├── components # 公共组件
├── directives # 自定义指令
├── layout # 布局组件
├── main.ts # 入口文件
├── plugins # 插件
├── router # 路由
├── stores # 状态管理
├── test # 单元测试
├── types # 类型定义
├── utils # 工具函数
└── views # 页面组件
